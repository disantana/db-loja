// REFERE-SE AS SERIES DISPONIVEIS NA PLATAFORMA
Table SERIES as S {
  COD_SERIES INT [pk, increment] // auto-increment
  NOME_SERIE VARCHAR(100) [not null]
  DESCRICAO_SERIE VARCHAR(500)
  STATUS CHAR
  ANO_LANCAMENTO TINYINT
  //QTD_TEMPORADA TINYINT
  //COD_CATEGORIA INT [ref: < CAT.COD_CATEGORIA]
}

Table CATEGORIA as CAT {
  COD_CATEGORIA INT [pk, increment]
  DESCRICAO  VARCHAR(30)
  STATUS CHAR
  //COD_CATEGORIA INT [ref: > C.COD_CATEGORIA]
}

Table SERIE_CATEGORIA as SC {
  COD_CATEGORIA INT [ref: > CAT.COD_CATEGORIA]
  COD_SERIE INT [ref: > S.COD_SERIES]
}

// REFERE-SE AOS CLIENTES DA PLATAFORMA
Table CLIENTES as C {
  COD_CLIENTE INT [pk, increment] // auto-increment
  NOME_CLIENTE VARCHAR(150) [not null]
  STATUS CHAR
  TITULAR INT [ref: > C.COD_CLIENTE, NOT NULL]
  CPF VARCAR(11) [NOT NULL, UNIQUE]
  FORMA_PAGAMENTO INT [ref: - FP.COD_FPAGAMENTO]
  //COD_CATEGORIA INT [ref: > C.COD_CATEGORIA]
}

Table AUDIENCIA as A {
  COD_AUDIENCIA INT [pk, increment]
  COD_CLIENTE INT [ref: > C.COD_CLIENTE]
  COD_EPISODIO  INT [ref: > E.COD_EPISODIO]
  DATA_ACESSO DATETIME [NOT NULL]
  TEMPO_CORRIDO BIGINT [NOT NULL]
  //COD_CATEGORIA INT [ref: > C.COD_CATEGORIA]
}

Table TEMPORADA as T {
  COD_TEMPORADA INT [pk, increment]
  DESCRICAO  VARCHAR(30)
  STATUS CHAR
  COD_SERIE INT [ref: > S.COD_SERIES]
  LANCAMENTO CHAR
  QTD_EPISODIO TINYINT
}

Table EPISODIO as E {
  COD_EPISODIO INT [pk, increment]
  DESCRICAO  VARCHAR(30)
  STATUS CHAR
  DURACAO BIGINT // EM SEG
  COD_VIDEO INT [ref: - V.COD_VIDEO]
  PERMITE_DOWNLOAD CHAR
  COD_TEMPORADA INT [ref: > T.COD_TEMPORADA]
}

Table VIDEO as V {
  COD_VIDEO INT [pk, increment]
  FILE_VIDEO BLOB
  //COD_CATEGORIA INT [ref: > C.COD_CATEGORIA]
}

Table FORMA_PAGAMENTO as FP {
  COD_FPAGAMENTO INT [pk, increment]
  COD_CLIENTE INT  [ref: > C.COD_CLIENTE]
  TIPO_PAGAMENTO CHAR
  COD_CARTAO BIGINT [ref: > CT.COD_CARTAO]

}
Table CARTAO as CT {
  COD_CARTAO BIGINT [pk, increment]
  NOME_CARTAO VARCHAR [NOT NULL]
  NUM_CARTAO BIGINT [NOT NULL]
  VALIDADE_CARTAO DATE [NOT NULL]
  CV_CARTAO INT [NOT NULL]
  COD_CLIENTE INT  [ref: - C.COD_CLIENTE]
  
  

  //COD_CATEGORIA INT [ref: > C.COD_CATEGORIA]
}

Table PLANO as P {
  COD_PLANO INT [pk, increment]
  VALOR     NUMERIC
  STATUS    CHAR
  DESCRICAO  VARCHAR(30)
  //COD_CATEGORIA INT [ref: > C.COD_CATEGORIA]

  
  //COD_CATEGORIA INT [ref: > C.COD_CATEGORIA]
}

Table ASSINATURA as ASSINATURA {
  COD_ASSINATURA INT [pk, increment]
  COD_CLIENTE INT [ref: > C.COD_CLIENTE]
  INICIO_VIGENCIA DATE
